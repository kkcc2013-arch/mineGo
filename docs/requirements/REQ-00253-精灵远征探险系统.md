## 需求描述

实现精灵远征探险系统，允许玩家派遣精灵团队执行自动化探险任务。精灵将在后台完成探索，获得经验、道具、稀有物品奖励。远征时长、难度、奖励根据精灵等级、属性、团队配置动态计算。

### 核心功能
1. **远征任务生成** - 根据玩家等级、地理位置、精灵属性动态生成远征任务
2. **团队配置管理** - 支持组建多支远征队伍，每队最多 6 只精灵
3. **自动探险执行** - 后台自动执行探险，支持实时进度追踪
4. **奖励分发机制** - 探险完成后自动发放奖励，支持稀有道具掉落
5. **精灵疲劳系统** - 远征消耗精灵体力，需要休息恢复
6. **紧急召回功能** - 支持中途召回，但奖励会大幅减少

### 2. 后端服务实现

#### expedition-service 核心逻辑

```javascript
// backend/services/expedition-service/src/services/ExpeditionManager.js

const { v4: uuidv4 } = require('uuid');
const { db } = require('@shared/database');
const { eventBus } = require('@shared/eventBus');
const { logger } = require('@shared/logger');
const { cache } = require('@shared/cache');

class ExpeditionManager {
  constructor() {
    this.MISSION_TYPES = {
      EXPLORATION: 'exploration',
      RESCUE: 'rescue',
      TREASURE: 'treasure',
      RAID: 'raid'
    };
    
    this.MAX_CONCURRENT_EXPEDITIONS = 3;
    this.FATIGUE_RECOVERY_RATE = 10; // 每小时恢复 10 点体力
    this.MAX_FATIGUE = 100;
  }

  /**
   * 获取可用的远征任务列表
   */
  async getAvailableMissions(userId, filters = {}) {
    const cacheKey = `expeditions:missions:${userId}:${JSON.stringify(filters)}`;
    const cached = await cache.get(cacheKey);
    if (cached) return cached;

    const user = await db.users.findById(userId);
    
    const missions = await db.expedition_missions.findMany({
      where: {
        is_active: true,
        min_pokemon_level: { lte: user.level },
        OR: [
          { valid_from: null },
          { valid_from: { lte: new Date() } }
        ],
        OR: [
          { valid_until: null },
          { valid_until: { gte: new Date() } }
        ]
      },
      orderBy: [
        { difficulty: 'asc' },
        { duration_minutes: 'asc' }
      ]
    });

    // 根据玩家地理位置和天气过滤
    const filtered = await this.filterByLocationAndWeather(missions, user);

    await cache.set(cacheKey, filtered, 'EX', 300); // 缓存 5 分钟
    return filtered;
  }

  /**
   * 启动远征
   */
  async startExpedition(userId, missionId, pokemonIds, config = {}) {
    // 验证前置条件
    await this.validateExpeditionStart(userId, missionId, pokemonIds);

    const mission = await db.expedition_missions.findById(missionId);
    const pokemon = await this.fetchAndValidatePokemon(userId, pokemonIds);

    // 计算成功率
    const successRate = this.calculateSuccessRate(mission, pokemon, config);

    // 计算远征时长（考虑加速效果）
    const durationMinutes = this.calculateDuration(mission, pokemon);

    const expedition = await db.expeditions.create({
      id: uuidv4(),
      user_id: userId,
      mission_id: missionId,
      team_pokemon_ids: pokemonIds,
      team_config: config,
      expected_end_at: new Date(Date.now() + durationMinutes * 60 * 1000),
      calculated_success_rate: successRate,
      status: 'ongoing',
      progress_percentage: 0
    });

    // 更新精灵状态和体力
    await this.applyPokemonExpeditionStart(pokemonIds);

    // 发布远征开始事件
    await eventBus.emit('expedition.started', {
      expeditionId: expedition.id,
      userId,
      missionId,
      durationMinutes,
      successRate
    });

    // 调度远征完成任务
    await this.scheduleExpeditionCompletion(expedition.id, durationMinutes);

    logger.info('Expedition started', {
      expeditionId: expedition.id,
      userId,
      missionId,
      durationMinutes
    });

    return expedition;
  }

  /**
   * 计算远征成功率
   */
  calculateSuccessRate(mission, pokemon, config) {
    let baseRate = 50; // 基础成功率 50%

    // 精灵等级加成
    const avgLevel = pokemon.reduce((sum, p) => sum + p.level, 0) / pokemon.length;
    const levelBonus = Math.min((avgLevel - mission.min_pokemon_level) * 2, 20);
    baseRate += levelBonus;

    // 属性匹配加成
    const typeMatchBonus = this.calculateTypeMatchBonus(mission, pokemon);
    baseRate += typeMatchBonus;

    // 团队配置加成
    if (config.formation === 'balanced') baseRate += 5;
    if (config.tactics === 'cautious') baseRate += 10;

    // 精灵疲劳惩罚
    const fatiguePenalty = this.calculateFatiguePenalty(pokemon);
    baseRate -= fatiguePenalty;

    // 难度惩罚
    const difficultyPenalty = (mission.difficulty - 1) * 8;
    baseRate -= difficultyPenalty;

    return Math.max(10, Math.min(95, baseRate));
  }

  /**
   * 获取远征进度
   */
  async getExpeditionProgress(expeditionId) {
    const expedition = await db.expeditions.findById(expeditionId);
    if (!expedition) {
      throw new Error('Expedition not found');
    }

    if (expedition.status !== 'ongoing') {
      return expedition;
    }

    // 计算实时进度
    const elapsed = Date.now() - new Date(expedition.started_at).getTime();
    const total = new Date(expedition.expected_end_at).getTime() - new Date(expedition.started_at).getTime();
    const progress = Math.min((elapsed / total) * 100, 100);

    // 生成随机事件
    const events = await this.generateExpeditionEvents(expedition, progress);

    return {
      ...expedition,
      progress_percentage: progress,
      current_events: events,
      time_remaining: Math.max(0, total - elapsed)
    };
  }

  /**
   * 紧急召回远征
   */
  async recallExpedition(userId, expeditionId) {
    const expedition = await db.expeditions.findById(expeditionId);
    
    if (!expedition || expedition.user_id !== userId) {
      throw new Error('Expedition not found or unauthorized');
    }

    if (expedition.status !== 'ongoing') {
      throw new Error('Expedition is not ongoing');
    }

    // 计算已完成的进度
    const progress = await this.getExpeditionProgress(expeditionId);

    // 召回奖励大幅减少（只有原奖励的 30%）
    const rewards = await this.calculateRecallRewards(expedition, progress.progress_percentage);

    // 更新远征状态
    await db.expeditions.update(expeditionId, {
      status: 'recalled',
      actual_end_at: new Date(),
      rewards,
      progress_percentage: progress.progress_percentage
    });

    // 发放奖励
    await this.distributeRewards(userId, rewards);

    // 恢复精灵状态
    await this.applyPokemonExpeditionEnd(expedition.team_pokemon_ids, progress.progress_percentage);

    await eventBus.emit('expedition.recalled', {
      expeditionId,
      userId,
      rewards,
      progress: progress.progress_percentage
    });

    logger.info('Expedition recalled', {
      expeditionId,
      userId,
      progress: progress.progress_percentage
    });

    return { status: 'recalled', rewards };
  }

  /**
   * 完成远征（定时任务调用）
   */
  async completeExpedition(expeditionId) {
    const expedition = await db.expeditions.findById(expeditionId);
    
    if (expedition.status !== 'ongoing') {
      logger.warn('Expedition already completed', { expeditionId });
      return;
    }

    // 根据成功率判断是否成功
    const isSuccess = Math.random() * 100 < expedition.calculated_success_rate;

    // 计算奖励
    const rewards = isSuccess 
      ? await this.calculateSuccessRewards(expedition)
      : await this.calculateFailureRewards(expedition);

    // 更新远征状态
    await db.expeditions.update(expeditionId, {
      status: isSuccess ? 'completed' : 'failed',
      actual_end_at: new Date(),
      progress_percentage: 100,
      rewards: rewards.regular,
      rare_rewards: rewards.rare
    });

    // 发放奖励
    if (rewards.regular.length > 0) {
      await this.distributeRewards(expedition.user_id, rewards);
    }

    // 记录精灵远征历史
    await this.recordPokemonHistory(expedition, rewards);

    // 恢复精灵状态
    await this.applyPokemonExpeditionEnd(expedition.team_pokemon_ids, 100);

    await eventBus.emit('expedition.completed', {
      expeditionId,
      userId: expedition.user_id,
      isSuccess,
      rewards
    });

    logger.info('Expedition completed', {
      expeditionId,
      isSuccess,
      rewardCount: rewards.regular.length
    });
  }

  /**
   * 计算成功奖励
   */
  async calculateSuccessRewards(expedition) {
    const mission = await db.expedition_missions.findById(expedition.mission_id);
    const rewards = [];

    // 从奖励池随机抽取
    for (const rewardDef of mission.reward_pool) {
      if (Math.random() < rewardDef.rarity) {
        const variance = rewardDef.amount_variance || 0.2;
        const amount = Math.round(rewardDef.amount * (1 + (Math.random() * 2 - 1) * variance));
        
        rewards.push({
          type: rewardDef.type,
          amount,
          rarity: rewardDef.rarity
        });
      }
    }

    // 稀有奖励抽取
    const rareRewards = [];
    if (mission.rare_reward_pool && Math.random() < 0.1) {
      const rareReward = this.selectRandomReward(mission.rare_reward_pool);
      rareRewards.push(rareReward);
    }

    return { regular: rewards, rare: rareRewards };
  }

  /**
   * 调度远征完成
   */
  async scheduleExpeditionCompletion(expeditionId, durationMinutes) {
    // 使用 Redis 延迟队列或定时任务
    const delayMs = durationMinutes * 60 * 1000;
    
    // 发布延迟事件
    await eventBus.emit('expedition.schedule_completion', {
      expeditionId,
      completeAt: Date.now() + delayMs
    }, { delay: delayMs });

    // 同时创建定时任务作为备份
    await db.scheduled_tasks.create({
      task_type: 'expedition_completion',
      task_data: { expeditionId },
      execute_at: new Date(Date.now() + delayMs)
    });
  }
}

module.exports = ExpeditionManager;
```

// backend/services/expedition-service/src/routes/expedition.js

const express = require('express');
const router = express.Router();
const ExpeditionManager = require('../services/ExpeditionManager');
const { auth } = require('@shared/middleware/auth');
const { validate } = require('@shared/middleware/validation');
const { rateLimit } = require('@shared/middleware/rateLimit');

const expeditionManager = new ExpeditionManager();

/**
 * GET /api/expeditions/missions
 * 获取可用远征任务列表
 */
router.get('/missions', auth, rateLimit({ windowMs: 60000, max: 30 }), async (req, res) => {
  try {
    const { difficulty, type, maxDuration } = req.query;
    const missions = await expeditionManager.getAvailableMissions(req.user.id, {
      difficulty,
      type,
      maxDuration
    });
    res.json({ success: true, data: missions });
  } catch (error) {
    res.status(500).json({ success: false, error: error.message });
  }
});

/**
 * POST /api/expeditions/start
 * 启动远征
 */
router.post('/start', auth, validate({
  missionId: { type: 'string', required: true },
  pokemonIds: { type: 'array', minItems: 1, maxItems: 6, required: true },
  config: { type: 'object', optional: true }
}), async (req, res) => {
  try {
    const { missionId, pokemonIds, config } = req.body;
    const expedition = await expeditionManager.startExpedition(
      req.user.id,
      missionId,
      pokemonIds,
      config
    );
    res.json({ success: true, data: expedition });
  } catch (error) {
    res.status(400).json({ success: false, error: error.message });
  }
});

/**
 * GET /api/expeditions/:id/progress
 * 获取远征进度
 */
router.get('/:id/progress', auth, async (req, res) => {
  try {

    res.status(404).json({ success: false, error: error.message });

// frontend/game-client/src/components/ExpeditionPanel.js

import React, { useState, useEffect } from 'react';
import { useTranslation } from 'react-i18next';
import './ExpeditionPanel.css';

const ExpeditionPanel = ({ userId }) => {
  const { t } = useTranslation();
  const [missions, setMissions] = useState([]);

  const [selectedMission, setSelectedMission] = useState(null);

  const [loading, setLoading] = useState(false);

  useEffect(() => {
    fetchMissions();
    fetchActiveExpeditions();
  }, [userId]);

  const fetchMissions = async () => {
    const response = await fetch('/api/expeditions/missions', {
      headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` }
    });
    const data = await response.json();
    if (data.success) setMissions(data.data);
  };

  const fetchActiveExpeditions = async () => {
    const response = await fetch('/api/expeditions/active', {
      headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` }
    });
    const data = await response.json();
    if (data.success) {
      setActiveExpeditions(data.data);
      // 设置进度轮询
      data.data.forEach(exp => startProgressPolling(exp.id));
    }
  };

  const startProgressPolling = (expeditionId) => {
    const interval = setInterval(async () => {
      const response = await fetch(`/api/expeditions/${expeditionId}/progress`, {
        headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` }
      });
      const data = await response.json();
      
      if (data.success && data.data.status !== 'ongoing') {
        clearInterval(interval);
        // 显示完成通知
        showCompletionNotification(data.data);
        fetchActiveExpeditions();
      } else if (data.success) {
        updateExpeditionProgress(expeditionId, data.data);
      }
    }, 10000); // 每 10 秒更新一次

    return () => clearInterval(interval);
  };

  const startExpedition = async () => {
    if (selectedPokemon.length === 0) {
      alert(t('expedition.selectPokemon'));
      return;
    }

    setLoading(true);
    try {
      const response = await fetch('/api/expeditions/start', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${localStorage.getItem('token')}`
        },
        body: JSON.stringify({
          missionId: selectedMission.id,
          pokemonIds: selectedPokemon,
          config: { formation: 'balanced', tactics: 'cautious' }
        })
      });

      const data = await response.json();
      if (data.success) {
        setActiveExpeditions([...activeExpeditions, data.data]);
        startProgressPolling(data.data.id);
        setSelectedMission(null);
        setSelectedPokemon([]);
      }
    } finally {
      setLoading(false);
    }
  };

  const recallExpedition = async (expeditionId) => {
    if (!confirm(t('expedition.confirmRecall'))) return;

    const response = await fetch(`/api/expeditions/${expeditionId}/recall`, {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` }
    });

    const data = await response.json();
    if (data.success) {
      setActiveExpeditions(activeExpeditions.filter(e => e.id !== expeditionId));
      alert(t('expedition.recalled', { rewards: JSON.stringify(data.data.rewards) }));
    }
  };

  return (
    <div className="expedition-panel">
      <h2>{t('expedition.title')}</h2>
      
      {/* 活跃远征列表 */}
      <div className="active-expeditions">
        <h3>{t('expedition.active')}</h3>
        {activeExpeditions.map(exp => (
          <div key={exp.id} className="expedition-card ongoing">
            <h4>{exp.mission?.title || 'Unknown Mission'}</h4>
            <div className="progress-bar">
              <div 
                className="progress-fill" 
                style={{ width: `${exp.progress_percentage || 0}%` }}
              />
            </div>
            <div className="time-remaining">
              {formatTimeRemaining(exp.expected_end_at)}
            </div>
            <button 
              className="recall-btn"
              onClick={() => recallExpedition(exp.id)}
            >
              {t('expedition.recall')}
            </button>
          </div>
        ))}
      </div>

      {/* 可用任务列表 */}
      <div className="available-missions">
        <h3>{t('expedition.available')}</h3>
        {missions.map(mission => (
          <div 
            key={mission.id} 
            className={`mission-card ${selectedMission?.id === mission.id ? 'selected' : ''}`}
            onClick={() => setSelectedMission(mission)}
          >
            <h4>{mission.title}</h4>
            <p>{mission.description}</p>
            <div className="mission-info">
              <span>⏱️ {mission.duration_minutes}min</span>
              <span>⭐ Difficulty: {mission.difficulty}</span>
              <span>🎯 Success Rate: ~{Math.round(70 - mission.difficulty * 8)}%</span>
            </div>
          </div>
        ))}
      </div>

      {/* 精灵选择 */}
      {selectedMission && (
        <div className="pokemon-selection">
          <h3>{t('expedition.selectTeam')}</h3>
          <p>Selected: {selectedPokemon.length}/{selectedMission.max_team_size}</p>
          {/* PokemonSelector 组件 */}
          <button 
            className="start-btn"
            onClick={startExpedition}
            disabled={loading || selectedPokemon.length === 0}
          >
            {loading ? t('common.loading') : t('expedition.start')}
          </button>
        </div>
      )}
    </div>
  );
};

 * 处理远征完成

 */
async function processExpeditionCompletions() {
  try {
    const now = new Date();
    
    // 查找到期的远征
    const expeditions = await db.expeditions.findMany({
      where: {
        status: 'ongoing',
        expected_end_at: { lte: now }
      }
    });

    logger.info(`Processing ${expeditions.length} expedition completions`);

    for (const expedition of expeditions) {
      try {
        await expeditionManager.completeExpedition(expedition.id);
      } catch (error) {
        logger.error('Failed to complete expedition', {
          expeditionId: expedition.id,
          error: error.message
        });
      }
    }
  } catch (error) {
    logger.error('Expedition completion processor error', { error: error.message });
  }
}

/**
 * 恢复精灵疲劳值
 * 每小时执行一次
 */
async function recoverPokemonFatigue() {
  try {
    const result = await db.$queryRaw`
      UPDATE pokemon
      SET fatigue = GREATEST(0, fatigue - 10),
          updated_at = NOW()
      WHERE fatigue > 0
      RETURNING id, fatigue
    `;

    logger.info(`Recovered fatigue for ${result.length} pokemon`);
  } catch (error) {
    logger.error('Fatigue recovery error', { error: error.message });
  }
}

// 导出任务
module.exports = {
  processExpeditionCompletions,
  recoverPokemonFatigue
};

// 如果直接运行
if (require.main === module) {
  processExpeditionCompletions()
    .then(() => process.exit(0))
    .catch(err => {
      logger.error('Job failed', { error: err.message });
      process.exit(1);
    });
}
